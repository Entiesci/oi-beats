int ksm(int c,int k,int p) {
//  p=MOD;
    if(!c)return 0;
    int res=1;
    while(k){
        if(k&1)res=(res*c)%p;
        c=(c*c)%p;k>>=1;
    }
    return res;
}

    fac[0]=1;
    for (int i=1;i<=n;++i) fac[i]=fac[i-1]*i%MOD;
    inv[n]=ksm(fac[n],MOD-2,MOD);
    for(int i=n;i>=1;i--)inv[i-1]=inv[i]*i%MOD;//i!的inv 
}

}

  if(n<0||m<0)return 0;
    if(n<m)return 0;
    return fac[n]*inv[m]%MOD*inv[n-m]%MOD;
}
